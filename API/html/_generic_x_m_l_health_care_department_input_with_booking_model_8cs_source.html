246  public DateTime NextDispatching(DateTime time)
247  {
248  DayOfWeek currentDay = time.DayOfWeek;
249 
250  int totalDaysPassed = 0;
251 
252  if (!(WaitingListDispatchTimesPerWeekday.ContainsKey(currentDay)
253  && time.Date + WaitingListDispatchTimesPerWeekday[currentDay].Last() > time))
254  {
255  currentDay = ParseEnum<DayOfWeek>((((int)(currentDay + 1) % 7)).ToString());
256  totalDaysPassed++;
257 
258  while (!WaitingListDispatchTimesPerWeekday.ContainsKey(currentDay) && totalDaysPassed <= 7)
259  {
260  currentDay = ParseEnum<DayOfWeek>((((int)(currentDay + 1)%7)).ToString());
261  totalDaysPassed++;
262  } // end while
263  } // end if
264 
265  if (totalDaysPassed == 0)
266  {
267  TimeSpan currentTime = WaitingListDispatchTimesPerWeekday[currentDay].Last();
268 
269  int currentIndex = WaitingListDispatchTimesPerWeekday[currentDay].Count - 1;
270 
271  while (time.Date + currentTime > time && currentIndex >= 1)
272  {
273  currentIndex--;
274  currentTime = WaitingListDispatchTimesPerWeekday[currentDay][currentIndex];
275  } // end while
276 
277  return time.Date + currentTime;
278  }
279  else
280  {
281  return time.Date + TimeSpan.FromDays(totalDaysPassed) + WaitingListDispatchTimesPerWeekday[currentDay].First();
282  } // end if
283 
284  } // end of LatestFollowUp
