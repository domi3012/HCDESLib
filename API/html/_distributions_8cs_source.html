295  public double TriangularDistribution(double lower, double peak, double upper)
296  {
297  double rand = RandomNumberGenerator.NextDouble();
298 
299  if (upper == lower)
300  return lower;
301 
302  if (rand <= (peak - lower) / (upper - lower))
303  return lower + Math.Sqrt(rand * (upper - lower) * (peak - lower));
304  else
305  return upper - Math.Sqrt((1 - rand) * (upper - lower) * (upper - peak));
