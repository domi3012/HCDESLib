732  public void GetCorePath(EntityPatient patient,
733  AdmissionType admissionType,
734  out List<ActionTypeClass> actions,
735  out Admission outpatientAdmission,
736  out Admission inpatientAdmission)
737  {
738  XMLPatientClass selectedPatientClass = null;
739 
740  foreach (XMLPatientClass xmlPatientClass in PatientClassPerXmlPatientClass.Keys)
741  {
742  if ((admissionType == null
743  || xmlPatientClass.AdmissionType == admissionType.Identifier
744  || !ConsiderAdmissionForPath)
745  && (patient.PatientClass.Equals(PatientClassPerXmlPatientClass[xmlPatientClass])
746  || !ConsiderPatietnClassForPath))
747  {
748  selectedPatientClass = xmlPatientClass;
749  break;
750  } // end if
751  } // end foreach
752 
753  if (selectedPatientClass == null)
754  throw new InvalidOperationException("Patient class and admission not found in XML-Patient Classes");
755 
756  actions = new List<ActionTypeClass>();
757 
758  XMLPatientPath selectedPath = null;
759 
760  if (selectedPatientClass.SinglePath >= 0)
761  {
762  selectedPath = XMLPathsPerID[selectedPatientClass.SinglePath];
763  }
764  else
765  {
766  selectedPath = XMLPathsPerID[selectedPatientClass.PathDistribution.GetRandomValue()];
767  } // end if
768 
769  foreach (XMLPatientPathStep pathStep in selectedPath.PathSteps)
770  {
771  if (pathStep.SingleAction != null)
772  {
773  actions.Add(ActionTypes[pathStep.SingleAction]);
774  }
775  else
776  {
777  bool sampleFound;
778  string actionType = pathStep.StepDistribution.GetNullableRandomValue(out sampleFound);
779 
780  if (!sampleFound)
781  continue;
782 
783  actions.Add(ActionTypes[(string)actionType]);
784  } // end if
785  } // end foreach
786 
787  outpatientAdmission = null;
788  inpatientAdmission = null;
789 
790  if (selectedPath.OutpatientAdmissions.Count > 0)
791  {
792  bool sampleFound;
793  XMLAdmission selectedAdmission = selectedPath.OutpatientAdmissionDistribution.GetNullableRandomValue(out sampleFound);
794 
795  if (sampleFound)
796  {
797  outpatientAdmission = new Admission(patient,
798  new OutpatientAdmissionTypes(selectedAdmission.AdmissionType),
799  selectedAdmission.MinDays,
800  selectedAdmission.MaxDays,
801  false,
802  patient.CorrespondingDoctor);
803  } // end if
804  } // end if
805 
806  if (selectedPath.InpatientAdmissions.Count > 0)
807  {
808  bool sampledFound;
809  XMLAdmission selectedAdmission = selectedPath.InpatientAdmissionDistribution.GetNullableRandomValue(out sampledFound);
810 
811  if (sampledFound)
812  {
813  inpatientAdmission = new Admission(patient,
814  new InpatientAdmissionTypes(selectedAdmission.AdmissionType,
815  false),
816  selectedAdmission.MinDays,
817  selectedAdmission.MaxDays,
818  false,
819  patient.CorrespondingDoctor);
820  } // end if
821  } // end if
822 
823  } // end of GetCorePath
