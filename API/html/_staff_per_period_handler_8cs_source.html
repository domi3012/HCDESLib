383  {

410  public Event GetNextStaffChangingEvent(ControlUnitHealthCareDepartment parentControl, DateTime time, out DateTime timeToOccur)
411  {
412  timeToOccur = time;
413 
414  // look for change later on the same day
415  List<TimeSpan> changesOfCurrentDay = SortedChangeTimesPerDay[time.DayOfWeek];
416  StaffChangingPoint closestChangingPoint = null;
417  StaffChangingPoint pointOnNextDayWithSameTime = null;
418 
419  if (changesOfCurrentDay.Count > 0 && changesOfCurrentDay.Last() > time.TimeOfDay)
420  {
421  for (int i = 0; i < changesOfCurrentDay.Count; i++)
422  {
423  if (changesOfCurrentDay[i] > time.TimeOfDay)
424  {
425  closestChangingPoint = StaffChangingPerDayAndTime[time.DayOfWeek][changesOfCurrentDay[i]];
426  timeToOccur = time.Date + closestChangingPoint.TimeOfDay;
427  break;
428  } // end if
429  } // end for
430 
431  // check if changing point was at end of day
432  if (closestChangingPoint.TimeOfDay.Days == 1)
433  {
434  DayOfWeek nextDay = (time.Date + TimeSpan.FromDays(1)).DayOfWeek;
435 
436  //checking if there is a changing point at the beginning of the next day
437  if (StaffChangingPerDayAndTime[nextDay].ContainsKey(TimeSpan.FromHours(0)))
438  {
439  pointOnNextDayWithSameTime = StaffChangingPerDayAndTime[nextDay][TimeSpan.FromHours(0)];
440  } // end if
441 
442  } // end if
443  }
444  else
445  {
446  DateTime nextTime = time.Date;
447 
448  while (closestChangingPoint == null)
449  {
450  nextTime += TimeSpan.FromDays(1);
451 
452  List<TimeSpan> changesOfNextDay = SortedChangeTimesPerDay[nextTime.DayOfWeek];
453 
454  if (changesOfNextDay.Count > 0)
455  {
456  closestChangingPoint = StaffChangingPerDayAndTime[nextTime.DayOfWeek][changesOfNextDay.First()];
457  timeToOccur = nextTime + closestChangingPoint.TimeOfDay;
458  } // end if
459 
460  } // end while
461  } // end if
462 
463  List<ResourceAssignmentStaff> allStaffLeaving = new List<ResourceAssignmentStaff>();
464  List<ResourceAssignmentStaff> allStaffArriving = new List<ResourceAssignmentStaff>();
465 
466  foreach (StaffAvailabilityPeriod period in closestChangingPoint.PeriodsEnding)
467  {
468  allStaffLeaving.AddRange(StaffPerPeriod[closestChangingPoint.DayOfWeek][period]);
469  } // end foreach
470 
471  foreach (StaffAvailabilityPeriod period in closestChangingPoint.PeriodsStarting)
472  {
473  allStaffArriving.AddRange(StaffPerPeriod[closestChangingPoint.DayOfWeek][period]);
474  } // end foreach
475 
476  // staff change on the beginning of next day
477  if (pointOnNextDayWithSameTime != null)
478  {
479  foreach (StaffAvailabilityPeriod period in pointOnNextDayWithSameTime.PeriodsEnding)
480  {
481  allStaffLeaving.AddRange(StaffPerPeriod[pointOnNextDayWithSameTime.DayOfWeek][period]);
482  } // end foreach
483 
484  foreach (StaffAvailabilityPeriod period in pointOnNextDayWithSameTime.PeriodsStarting)
485  {
486  allStaffArriving.AddRange(StaffPerPeriod[pointOnNextDayWithSameTime.DayOfWeek][period]);
487  } // end foreach
488  } // end if
489 
490  List<EntityHealthCareStaff> arrivingStaff = allStaffArriving.Select(p => p.Resource).ToList();
491  List<EntityHealthCareStaff> leavingStaff = allStaffLeaving.Select(p => p.Resource).ToList();
492  List<EntityHealthCareStaff> jointStaff = arrivingStaff.Where(p => leavingStaff.Contains(p)).ToList();
493 
494  return new EventStaffChange(parentControl,
495  allStaffLeaving.Where(p=> !jointStaff.Contains(p.Resource)).ToArray(),
496  allStaffArriving.Where(p => !jointStaff.Contains(p.Resource)).ToArray(),
497  this);
498 
499  } // end of GetNextStaffChangingEvent
