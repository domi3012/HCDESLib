226  public DrawingObject DrawingObjectPerEntity(Entity entity)
227  {
228  // if entity drawing object hasn't been created
229  // the corresponding method is looked up
230  if (!_drawingObjectPerEntity.ContainsKey(entity))
231  {
232  // in case no method to create a drawing object for the entity is provided an
233  // exception is thrown
234  if(!EntityVisualizationObjectCreatingMethodsPerType.ContainsKey(entity.GetType()))
235  throw new InvalidOperationException(String.Format("Creation of Drawing Object for Entity {0} is not provided", entity.ToString()));
236 
237  // create the drawing object
238  DrawingObject newDrawingObject = EntityVisualizationObjectCreatingMethodsPerType[entity.GetType()](entity);
239 
240  // add drawing obecjt to system, has to be set to visual manually by user
241  DrawingSystem.AddObject(newDrawingObject);
242 
243  // drawing object is stored in the dictionary
244  _drawingObjectPerEntity.Add(entity, newDrawingObject);
245 
246  } // end if
247 
248  // the right key is now in the dictionary
249  return _drawingObjectPerEntity[entity];
250  } // end of DrawingObjectPerEntity

285  public void VisualizeDynamicModel(DateTime currentTime, SimulationModel simModel, ControlUnit parentControlUnit, IEnumerable<Event> currentEvents)
286  {
287  #region VisualizationChangesFromEvents
288 
289  // visualization methods for all events are called (if provided)
290  foreach (EventActivity ev in currentEvents.Where(p=> p is EventActivity && ((EventActivity)p).EventType == EventType.End).Cast<EventActivity>())
291  {
292  Activity parentActivity = ((EventActivity)ev).ParentActivity;
293 
294  if (parentActivity.StartTime == parentActivity.EndTime)
295  continue;
296 
297  // if a handling method for the end event is specified it is called
298  if (ActivityEndEventVisualizationMethods.ContainsKey(parentActivity.GetType()))
299  ActivityEndEventVisualizationMethods[parentActivity.GetType()](parentActivity, currentTime);
300  } // end foreach
301 
302  // visualization methods for all events are called (if provided)
303  foreach (EventActivity ev in currentEvents.Where(p=> p is EventActivity && ((EventActivity)p).EventType == EventType.Start).Cast<EventActivity>())
304  {
305  Activity parentActivity = ((EventActivity)ev).ParentActivity;
306 
307  if (parentActivity.StartTime == parentActivity.EndTime)
308  continue;
309 
310  // if a handling method for the start event is specified it is called
311  if (ActivityStartEventVisualizationMethods.ContainsKey(parentActivity.GetType()))
312  ActivityStartEventVisualizationMethods[parentActivity.GetType()](parentActivity, currentTime);
313  } // end foreach
314 
315  // visualization methods for all events are called (if provided)
316  foreach (Event ev in currentEvents.Where(p => p.EventType == EventType.Standalone))
317  {
318  // if a handling method for a standalone event is specified it is called
319  if (EventStandaloneDrawingMethods.ContainsKey(ev.GetType()))
320  EventStandaloneDrawingMethods[ev.GetType()](ev);
321  } // end foreach
322 
323  #endregion
324 
325  #region VisualizationDuringActivities
326 
327  foreach (Activity activity in parentControlUnit.CurrentActivities)
328  {
329  // intermediate visualization is only done for times strictly larger
330  // than the start time
331  if (activity.StartTime == currentTime)
332  continue;
333 
334  // if a method has been specified for an activity type it is called
335  if (ActivityIntermediateVisualizationMethods.ContainsKey(activity.GetType()))
336  ActivityIntermediateVisualizationMethods[activity.GetType()](activity, currentTime);
337 
338  } // end foreach
339 
340  #endregion
341 
342  #region VisualizeDynamicHoldingEntities
343 
344  foreach (IDynamicHoldingEntity holdingEntity in parentControlUnit.AllControlledEntities.Where(p => p is IDynamicHoldingEntity).Cast<IDynamicHoldingEntity>())
345  {
346  if (HoldingEntitiesVisualizationMethods.ContainsKey(holdingEntity.GetType()))
347  HoldingEntitiesVisualizationMethods[holdingEntity.GetType()](holdingEntity);
348  } // end foreach
349 
350  #endregion

352  AdditionalDynamicVisualization(currentTime, simModel, parentControlUnit, currentEvents);
353 
354  } // end of VisualizeDynamicModel
