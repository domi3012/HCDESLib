201  public bool RunSingleStepSimulationModel(DateTime currentTime, out DateTime newTime)
202  {

204  _currentlyTriggeredEvents.Clear();

206  newTime = currentTime;
207 
208  // all scheduled behavior of this time is executed
209  if (SELlist.NumberEvents == 0)
210  {
211  newTime = SimulationModel.EndTime;
212  return false;

214  else
215  {
216  while (SELlist.GetFirstEventTime() == currentTime)

218  SELlist.TriggerFirstEvent(this);
219  } // end while
220  } // end if
221 
222  // perform conditional rules on model
223  SimulationModel.PerformModelRules(CurrentlyTriggeredEvents, currentTime, this);
224
